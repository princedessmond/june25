function testPoliceAlert() {
  Logger.log('🧪 Testing Police Alert submission...');

  var testData = {
    postData: {
      contents: JSON.stringify({
        id: 'POL' + Date.now(),
        timestamp: new Date().toISOString(),
        county: 'nairobi',
        specificLocation: 'Uhuru Park - APPS SCRIPT TEST',
        activityType: 'large-deployment',
        officerCount: '20-50',
        vehicleCount: '5-10',
        description: 'TEST from Google Apps Script Editor - Please ignore',
        reporterContact: '0712345678',
        latitude: '-1.2864',
        longitude: '36.8172',
        gpsAccuracy: '50'
      })
    }
  };

  Logger.log('📤 Sending test data...');

  try {
    var result = doPost(testData);
    var response = JSON.parse(result.getContent());

    Logger.log('✅ Test completed!');
    Logger.log('📊 Result: ' + JSON.stringify(response));

    if (response.result === 'success') {
      Logger.log('🎉 SUCCESS! Data was added to row: ' + response.row);
      Logger.log('📋 Check your "Police Alerts" sheet now!');
    } else {
      Logger.log('❌ ERROR: ' + response.error);
    }
  } catch (error) {
    Logger.log('❌ TEST FAILED: ' + error.toString());
    Logger.log('💡 Check that:');
    Logger.log('   1. The doPost function exists above this test function');
    Logger.log('   2. The "Police Alerts" sheet exists in your spreadsheet');
    Logger.log('   3. You have edit permissions on this spreadsheet');
  }
}
