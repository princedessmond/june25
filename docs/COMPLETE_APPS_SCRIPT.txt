function doPost(e) {
  try {
    // Parse the incoming data
    var data = JSON.parse(e.postData.contents);

    // Log the received data for debugging
    Logger.log('Received data: ' + JSON.stringify(data));
    Logger.log('Data ID: ' + data.id);

    // Get the spreadsheet
    var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();

    // Determine which sheet to use based on data type
    var sheetName;
    if (data.id && data.id.startsWith('REG')) {
      sheetName = 'Registrations';
    } else if (data.id && data.id.startsWith('TRN')) {
      sheetName = 'Transport Requests';
    } else if (data.id && data.id.startsWith('LEG')) {
      sheetName = 'Legal Aid Requests';
    } else if (data.id && data.id.startsWith('POL')) {
      sheetName = 'Police Alerts';
    } else {
      sheetName = 'Registrations'; // Default
    }

    Logger.log('Target sheet: ' + sheetName);

    var sheet = spreadsheet.getSheetByName(sheetName);

    // If sheet doesn't exist, create it with headers
    if (!sheet) {
      sheet = spreadsheet.insertSheet(sheetName);
      Logger.log('Created new sheet: ' + sheetName);

      // Add headers based on sheet type
      if (sheetName === 'Registrations') {
        sheet.appendRow(['Timestamp', 'Reference ID', 'Full Name', 'Phone', 'Email', 'County', 'Assembly Point', 'Need Transport']);
      } else if (sheetName === 'Transport Requests') {
        sheet.appendRow(['Timestamp', 'Reference ID', 'Name', 'Phone', 'Departure Point', 'Passenger Count', 'Route']);
      } else if (sheetName === 'Legal Aid Requests') {
        sheet.appendRow(['Timestamp', 'Case Reference', 'Name', 'Phone', 'Email', 'Urgency', 'Issue Description', 'Current Location']);
      } else if (sheetName === 'Police Alerts') {
        sheet.appendRow(['Timestamp', 'Alert ID', 'County', 'Specific Location', 'Activity Type', 'Officer Count', 'Vehicle Count', 'Description', 'Reporter Contact', 'Latitude', 'Longitude', 'GPS Accuracy']);
      }
    }

    // Prepare the data row
    var rowData = [];

    if (sheetName === 'Registrations') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.fullName || '',
        data.phone || '',
        data.email || '',
        data.location || '',
        data.assemblyPoint || '',
        data.needTransport ? 'Yes' : 'No'
      ];
    } else if (sheetName === 'Transport Requests') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.name || '',
        data.phone || '',
        data.departurePoint || '',
        data.passengerCount || '',
        data.route || ''
      ];
    } else if (sheetName === 'Legal Aid Requests') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.name || '',
        data.phone || '',
        data.email || '',
        data.urgency || '',
        data.issue || '',
        data.location || ''
      ];
    } else if (sheetName === 'Police Alerts') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.county || '',
        data.specificLocation || '',
        data.activityType || '',
        data.officerCount || '',
        data.vehicleCount || '',
        data.description || '',
        data.reporterContact || '',
        data.latitude || '',
        data.longitude || '',
        data.gpsAccuracy || ''
      ];
    }

    // Log the row data before adding
    Logger.log('Row data to add: ' + JSON.stringify(rowData));

    // Add the row to the sheet
    sheet.appendRow(rowData);

    var lastRow = sheet.getLastRow();
    Logger.log('Data added to row: ' + lastRow);

    // Return success
    return ContentService
      .createTextOutput(JSON.stringify({
        'result': 'success',
        'row': lastRow,
        'sheet': sheetName,
        'id': data.id
      }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    // Log the error
    Logger.log('ERROR: ' + error.toString());

    // Return error
    return ContentService
      .createTextOutput(JSON.stringify({
        'result': 'error',
        'error': error.toString()
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
