function doPost(e) {
  try {
    // Parse the incoming data
    var data = JSON.parse(e.postData.contents);

    // Log the received data for debugging
    Logger.log('Received data: ' + JSON.stringify(data));
    Logger.log('Data ID: ' + data.id);

    // Get the spreadsheet
    var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();

    // Determine which sheet to use based on data type
    var sheetName;
    if (data.id && data.id.startsWith('REG')) {
      sheetName = 'Registrations';
    } else if (data.id && data.id.startsWith('TRN')) {
      sheetName = 'Transport Requests';
    } else if (data.id && data.id.startsWith('LEG')) {
      sheetName = 'Legal Aid Requests';
    } else if (data.id && data.id.startsWith('POL')) {
      sheetName = 'Police Alerts';
    } else {
      sheetName = 'Registrations'; // Default
    }

    Logger.log('Target sheet: ' + sheetName);

    var sheet = spreadsheet.getSheetByName(sheetName);

    // If sheet doesn't exist, create it
    if (!sheet) {
      sheet = spreadsheet.insertSheet(sheetName);
      Logger.log('Created new sheet: ' + sheetName);
    }

    // Check if headers exist (row 1 should have content)
    var firstRow = sheet.getRange(1, 1, 1, 12).getValues()[0];
    var hasHeaders = firstRow.join('').length > 0;

    if (!hasHeaders) {
      Logger.log('No headers found, adding headers...');

      // Add headers based on sheet type
      if (sheetName === 'Registrations') {
        sheet.getRange(1, 1, 1, 8).setValues([
          ['Timestamp', 'Reference ID', 'Full Name', 'Phone', 'Email', 'County', 'Assembly Point', 'Need Transport']
        ]);
      } else if (sheetName === 'Transport Requests') {
        sheet.getRange(1, 1, 1, 7).setValues([
          ['Timestamp', 'Reference ID', 'Name', 'Phone', 'Departure Point', 'Passenger Count', 'Route']
        ]);
      } else if (sheetName === 'Legal Aid Requests') {
        sheet.getRange(1, 1, 1, 8).setValues([
          ['Timestamp', 'Case Reference', 'Name', 'Phone', 'Email', 'Urgency', 'Issue Description', 'Current Location']
        ]);
      } else if (sheetName === 'Police Alerts') {
        sheet.getRange(1, 1, 1, 12).setValues([
          ['Timestamp', 'Alert ID', 'County', 'Specific Location', 'Activity Type', 'Officer Count', 'Vehicle Count', 'Description', 'Reporter Contact', 'Latitude', 'Longitude', 'GPS Accuracy']
        ]);
      }

      // Format header row
      var headerRange = sheet.getRange(1, 1, 1, 12);
      headerRange.setFontWeight('bold');
      headerRange.setBackground('#4285f4');
      headerRange.setFontColor('#ffffff');
    }

    // Prepare the data row
    var rowData = [];

    if (sheetName === 'Registrations') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.fullName || '',
        data.phone || '',
        data.email || '',
        data.location || '',
        data.assemblyPoint || '',
        data.needTransport ? 'Yes' : 'No'
      ];
    } else if (sheetName === 'Transport Requests') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.name || '',
        data.phone || '',
        data.departurePoint || '',
        data.passengerCount || '',
        data.route || ''
      ];
    } else if (sheetName === 'Legal Aid Requests') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.name || '',
        data.phone || '',
        data.email || '',
        data.urgency || '',
        data.issue || '',
        data.location || ''
      ];
    } else if (sheetName === 'Police Alerts') {
      rowData = [
        new Date(data.timestamp),
        data.id,
        data.county || '',
        data.specificLocation || '',
        data.activityType || '',
        data.officerCount || '',
        data.vehicleCount || '',
        data.description || '',
        data.reporterContact || '',
        data.latitude || '',
        data.longitude || '',
        data.gpsAccuracy || ''
      ];
    }

    // Log the row data before adding
    Logger.log('Row data to add: ' + JSON.stringify(rowData));

    // Add the row to the sheet
    sheet.appendRow(rowData);

    var lastRow = sheet.getLastRow();
    Logger.log('Data added to row: ' + lastRow);

    // Return success
    return ContentService
      .createTextOutput(JSON.stringify({
        'result': 'success',
        'row': lastRow,
        'sheet': sheetName,
        'id': data.id
      }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    // Log the error
    Logger.log('ERROR: ' + error.toString());

    // Return error
    return ContentService
      .createTextOutput(JSON.stringify({
        'result': 'error',
        'error': error.toString()
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// Function to clean up and reset Police Alerts sheet
function resetPoliceAlertsSheet() {
  var spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = spreadsheet.getSheetByName('Police Alerts');

  if (sheet) {
    // Clear all content
    sheet.clear();

    // Add headers
    sheet.getRange(1, 1, 1, 12).setValues([
      ['Timestamp', 'Alert ID', 'County', 'Specific Location', 'Activity Type', 'Officer Count', 'Vehicle Count', 'Description', 'Reporter Contact', 'Latitude', 'Longitude', 'GPS Accuracy']
    ]);

    // Format headers
    var headerRange = sheet.getRange(1, 1, 1, 12);
    headerRange.setFontWeight('bold');
    headerRange.setBackground('#4285f4');
    headerRange.setFontColor('#ffffff');

    Logger.log('✅ Police Alerts sheet reset with headers in row 1');
  } else {
    Logger.log('❌ Police Alerts sheet not found');
  }
}
